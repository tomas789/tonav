100 Publisher::Publisher(ros::NodeHandle& nh, const std::string& base_topic, uint32_t queue_size,
101  const SubscriberStatusCallback& connect_cb,
102  const SubscriberStatusCallback& disconnect_cb,
103  const ros::VoidPtr& tracked_object, bool latch,
104  const PubLoaderPtr& loader)
105  : impl_(new Impl)
106 {
107  // Resolve the name explicitly because otherwise the compressed topics don't remap
108  // properly (#3652).
109  impl_->base_topic_ = nh.resolveName(base_topic);
110  impl_->loader_ = loader;
111 
112  BOOST_FOREACH(const std::string& lookup_name, loader->getDeclaredClasses()) {
113  try {
114  boost::shared_ptr<PublisherPlugin> pub = loader->createInstance(lookup_name);
115  impl_->publishers_.push_back(pub);
116  pub->advertise(nh, impl_->base_topic_, queue_size, rebindCB(connect_cb),
117  rebindCB(disconnect_cb), tracked_object, latch);
118  }
119  catch (const std::runtime_error& e) {
120  ROS_DEBUG("Failed to load plugin %s, error string: %s",
121  lookup_name.c_str(), e.what());
122  }
123  }
124 
125  if (impl_->publishers_.empty())
126  throw Exception("No plugins found! Does `rospack plugins --attrib=plugin "
127  "image_transport` find any packages?");
128 }

189 SubscriberStatusCallback Publisher::rebindCB(const SubscriberStatusCallback& user_cb)
190 {
191  // Note: the subscriber callback must be bound to the internal Impl object, not
192  // 'this'. Due to copying behavior the Impl object may outlive the original Publisher
193  // instance. But it should not outlive the last Publisher, so we use a weak_ptr.
194  if (user_cb)
195  {
196  ImplWPtr impl_wptr(impl_);
197  return boost::bind(&Publisher::weakSubscriberCb, impl_wptr, _1, user_cb);
198  }
199  else
200  return SubscriberStatusCallback();
201 }
